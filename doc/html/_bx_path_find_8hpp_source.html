446  for(int i = 0; i < Dst.count; ++i)
447  {
448  if(IsL) IsL &= (Dst.pt[i].x < SrcBound.l);
449  if(IsT) IsT &= (Dst.pt[i].y < SrcBound.t);
450  if(IsR) IsR &= (Dst.pt[i].x > SrcBound.r);
451  if(IsB) IsB &= (Dst.pt[i].y > SrcBound.b);
452  if(!IsBoundLine && BestDstI == -1 && (Dst.pt[i].x < SrcBound.l || Dst.pt[i].y < SrcBound.t))
453  BestDstI = i;
454  }
455  BxAssert("BxPathFind<Src는 Dst의 우측하단에 위치하여야 안전합니다>", IsBoundLine || BestDstI != -1);
456  if(IsL || IsT || IsR || IsB) return nullptr;
457  // 교차검사 수집데이터(최소한 Dst의 0번 정점은 Src에 포함되지 않아야 함)
458  point CollectDstB[1024];
459  point CollectDstE[1024];

461  BxCore::Util::MemMove(CollectDstB, &Dst.pt[0], sizeof(point) * CollectDstLength);
462  BxCore::Util::MemMove(CollectDstE, &Dst.pt[1], sizeof(point) * (CollectDstLength - 1));
463  CollectDstE[CollectDstLength - 1] = Dst.pt[0];
464  point CollectSrcB[1024];
465  point CollectSrcE[1024];
466  int CollectSrcLength = Src.count;
467  BxCore::Util::MemMove(CollectSrcB, &Src.pt[0], sizeof(point) * CollectSrcLength);
468  BxCore::Util::MemMove(CollectSrcE, &Src.pt[1], sizeof(point) * (CollectSrcLength - 1));
469  CollectSrcE[CollectSrcLength - 1] = Src.pt[0];
470  for(int d = 0; d < CollectDstLength; ++d)
471  for(int s = 0; s < CollectSrcLength; ++s)
472  {
473  const point* CrossDot = nullptr;

475  {
476  if((CrossDot->x != CollectDstB[d].x || CrossDot->y != CollectDstB[d].y)
477  && (CrossDot->x != CollectDstE[d].x || CrossDot->y != CollectDstE[d].y))
478  {
479  CollectDstE[CollectDstLength] = CollectDstE[d];
480  CollectDstB[CollectDstLength++] = (CollectDstE[d] = *CrossDot);
481  }
482  if((CrossDot->x != CollectSrcB[s].x || CrossDot->y != CollectSrcB[s].y)
483  && (CrossDot->x != CollectSrcE[s].x || CrossDot->y != CollectSrcE[s].y))
484  {
485  CollectSrcE[CollectSrcLength] = CollectSrcE[s];
486  CollectSrcB[CollectSrcLength++] = (CollectSrcE[s] = *CrossDot);
487  }
488  BxAssert("BxPathFind<계산상 오류입니다>", CollectDstLength < 1024 && CollectSrcLength < 1024);
489  }
490  }
491  if(CollectDstLength == Dst.count)
492  return nullptr;
493  Result.count = 0;
494  if(IsBoundLine)
495  {
496  Result.pt[0] = point::zero();
497  for(int d = 0; d < CollectDstLength; ++d)
498  {
499  if(CollectDstB[d].x <= SrcBound.l) continue;
500  if(CollectDstB[d].y <= SrcBound.t) continue;

288  void Targeting(const int NumPart, const int RandKey)
289  {
290  BxUtil::SetRandom(RandKey);
291  const int Detail = 256;
292  const bool IsCircle = (Position < 18 && Position / 9 == 1);
293  const bool IsRandom = (Position < 18 && (Position % 9) / 3 == 1);
294  const bool IsPull = (Position == 19 || (Position % 9) / 3 == 2);
295  const bool IsMiddle = (18 <= Position);
296  int RateLeft = 0, RateTop = 0, RateRight = 0, RateBottom = 0;
297  // 메모리할당
298  if(Animation) BxCore::Util::Free(Animation);
299  AnimationLength = NumPart;
300  Animation = BxCore::Util::Malloc(sizeof(_PartAnimation) * AnimationLength);
301  BxCore::Util::MemSet(BxCore::Util::GetPtr(Animation), 0, sizeof(_PartAnimation) * AnimationLength);
302  id_memory Point = BxCore::Util::Malloc(sizeof(int) * AnimationLength * 2);
303  BxCore::Util::MemSet(BxCore::Util::GetPtr(Point), 0, sizeof(int) * AnimationLength * 2);
304  // 포인터화
305  _PartAnimation* AnimationPtr = (_PartAnimation*) BxCore::Util::GetPtr(Animation);
306  int* PosX = &((int*) BxCore::Util::GetPtr(Point))[0];
307  int* PosY = &((int*) BxCore::Util::GetPtr(Point))[AnimationLength];
308 
309  if(!IsMiddle)
310  {
311  // 수집과정
312  int PrevPosX = PosX[0];
313  int PrevPosY = PosY[0];
314  int Dist = Detail * 2, Angle = 0, Count = 1, Step = 0, Vector = 0;
315  for(int i = 1; i < AnimationLength; ++i)
316  {
317  if(IsCircle)
318  {
319  PosX[i] = FtoI(Dist * BxUtil::Cos(Angle));
320  PosY[i] = FtoI(Dist * BxUtil::Sin(Angle));
321  }
322  else
323  {
324  PosX[i] = PrevPosX;
325  PosY[i] = PrevPosY;
326  switch(Vector)
327  {
328  case 0: PosX[i] += Detail * 2; break;
329  case 1: PosY[i] += Detail * 2; break;
330  case 2: PosX[i] -= Detail * 2; break;
331  case 3: PosY[i] -= Detail * 2; break;
332  }
333  PrevPosX = PosX[i];
334  PrevPosY = PosY[i];
335  }
336  if(IsRandom)
337  {
338  PosX[i] = (PosX[i] - Detail * 4 + (BxUtil::GetRandom() % (Detail * 8))) * ((BxUtil::GetRandom() % 2)? 1 : -1);
339  PosY[i] = (PosY[i] - Detail * 4 + (BxUtil::GetRandom() % (Detail * 8))) * ((BxUtil::GetRandom() % 2)? 1 : -1);
340  }
341  else if(IsPull)
342  {
343  PosX[i] -= Detail / 2 - (BxUtil::GetRandom() % Detail);
344  PosY[i] -= Detail / 2 - (BxUtil::GetRandom() % Detail);
345  }
346  RateLeft = BxUtilGlobal::Min(RateLeft, PosX[i]);
347  RateTop = BxUtilGlobal::Min(RateTop, PosY[i]);
348  RateRight = BxUtilGlobal::Max(RateRight, PosX[i]);
349  RateBottom = BxUtilGlobal::Max(RateBottom, PosY[i]);
350  if(IsCircle)
351  {
352  BxAssert("BxTween", (Dist * 2 * BxUtilGlobal::Pi()) != 0);
353  const int AddAngle = (int)((((huge)(Detail * 2 * 1024)) << 16) / (Dist * 2 * BxUtilGlobal::Pi()));
354  Angle = (Angle + AddAngle) % 1024;
355  Dist += Detail * 2 * AddAngle / 1024;
356  }
357  else
358  {
359  if(++Step == Count)
360  {
361  Step = 0;
362  if(Vector % 2) ++Count;
363  Vector = (Vector + 1) % 4;
364  }
365  }
366  }
367  // 가공과정
368  const int CenterX = (RateLeft + RateRight) / 2, CenterY = (RateTop + RateBottom) / 2;
369  const int SumDistPow = (RateRight - RateLeft) * (RateRight - RateLeft)
370  + (RateBottom - RateTop) * (RateBottom - RateTop);
371  const int SumDist = (0 < SumDistPow)? BxUtilGlobal::Sqrt(SumDistPow) : 0;
372  for(int i = 0; i < AnimationLength; ++i)
373  {
374  PosX[i] -= CenterX;
375  PosY[i] -= CenterY;
376  if(0 < Position % 3)
377  {
378  const int DistPow = PosX[i] * PosX[i] + PosY[i] * PosY[i];
379  const int Dist = (0 < DistPow)? BxUtilGlobal::Sqrt(DistPow) : 0;
380  const huge CalcDist = (Position % 3 == 1)? Dist : SumDist - Dist;
381  BxAssert("BxTween", (AnimationLength * Detail) != 0);
382  PosX[i] = (int)(PosX[i] * CalcDist / (AnimationLength * Detail));
383  PosY[i] = (int)(PosY[i] * CalcDist / (AnimationLength * Detail));
384  }
385  RateLeft = (i == 0)? PosX[i] : BxUtilGlobal::Min(RateLeft, PosX[i]);
386  RateTop = (i == 0)? PosY[i] : BxUtilGlobal::Min(RateTop, PosY[i]);
387  RateRight = (i == 0)? PosX[i] : BxUtilGlobal::Max(RateRight, PosX[i]);
388  RateBottom = (i == 0)? PosY[i] : BxUtilGlobal::Max(RateBottom, PosY[i]);
389  }
390  }
391  // 비례과정
392  int EnableRate = (IsBegin)? 100 : BxUtil::GetValue(PartRateMin, PartRateMax);
393  int EnableRateSum = 100;
394  for(int i = 0; i < AnimationLength; ++i)
395  {
396  BxAssert("BxTween", (RateRight - RateLeft + 2) != 0);
397  BxAssert("BxTween", (RateBottom - RateTop + 2) != 0);
398  AnimationPtr[i].ResetTarget(100 <= EnableRateSum,
399  (PosX[i] - RateLeft + 1) * (TargetX2 - TargetX1 + 1) / (RateRight - RateLeft + 2) - (TargetX2 - TargetX1) / 2,
400  (PosY[i] - RateTop + 1) * (TargetY2 - TargetY1 + 1) / (RateBottom - RateTop + 2) - (TargetY2 - TargetY1) / 2);
401  EnableRateSum = (EnableRateSum % 100) + EnableRate;
402  }
403  BxCore::Util::Free(Point);
404  }

431  void PlayAnimation(_PartBegin* PrevTarget, int MsPerFrame)
432  {
433  #define MS_TO_FRAME(A) (((A) + MsPerFrame / 2) / MsPerFrame)
434  _PartAnimation* AnimationPtr = (_PartAnimation*) BxCore::Util::GetPtr(Animation);
435  for(int i = 0, j = 0; i < AnimationLength; ++i)
436  {
437  if(!AnimationPtr[i].IsEnable)
438  continue;
439  // 진행
440  if(AnimationPtr[i].GetRunAnimation())
441  {
442  const int EndFrame = BxUtilGlobal::Max(1, MS_TO_FRAME(BxUtil::GetValue(WorkTimeMin, WorkTimeMax, i)));
443  const int AniFrame = ++AnimationPtr[i].AnimationFrame;
444  if(AniFrame <= 0)
445  {
446  // 불투명도적용
447  const int Step = AnimationPtr[i].AnimationFrame - AnimationPtr[i].AnimationFrameBegin;
448  const int End = -AnimationPtr[i].AnimationFrameBegin;
449  switch(Waiting)
450  {
451  case 0: // 급진소멸
452  AnimationPtr[i].SetOpacity(0, 1, 1);
453  break;
454  case 1: // 완만소멸
455  if(PrevTarget->IsBegin)
456  AnimationPtr[i].SetOpacity(0, End - Step, End);
457  else AnimationPtr[i].SetOpacity(0, (Step <= End / 2)? Step : End - Step, End / 2);
458  break;
459  case 2: // 급진점화
460  AnimationPtr[i].SetOpacity(100, 1, 1);
461  break;
462  case 3: // 완만점화
463  if(PrevTarget->IsBegin)
464  AnimationPtr[i].SetOpacity(0, End - Step, End);
465  else AnimationPtr[i].SetOpacity(100, (Step <= End / 2)? Step : End - Step, End / 2);
466  break;
467  case 4: // 규칙점멸
468  AnimationPtr[i].SetOpacity(50 * BxUtilGlobal::Abs(((i + Step) % 4) - 2), 1, 1);
469  break;
470  case 5: // 랜덤점멸
471  AnimationPtr[i].SetOpacity(50 * BxUtilGlobal::Abs((BxUtil::GetRandom(i + Step) % 4) - 2), 1, 1);
472  break;
473  }
474  }
475  else if(AniFrame <= EndFrame)
476  {
477  // 이동계산
478  int EndValue = 0, AniValue = 0;
479  switch(Movement)
480  {
481  case 0: // 등속
482  EndValue = EndFrame;
483  AniValue = AniFrame;
484  break;
485  case 1: // 가속
486  EndValue = EndFrame * EndFrame;
487  AniValue = AniFrame * AniFrame;
488  break;
489  case 2: // 감속
490  EndValue = EndFrame * EndFrame;
491  AniValue = EndValue - (EndFrame - AniFrame) * (EndFrame - AniFrame);
492  break;
493  case 3: // 가감변속
494  if(AniFrame <= EndFrame / 2)
495  {
496  const int End = EndFrame / 2;
497  const int EndHalf = End * End;
498  EndValue = EndHalf * 2;
499  AniValue = AniFrame * AniFrame;
500  }
501  else
502  {
503  const int Ani = AniFrame - EndFrame / 2;
504  const int End = EndFrame - EndFrame / 2;
505  const int EndHalf = End * End;
506  EndValue = EndHalf * 2;
507  AniValue = EndHalf + EndHalf - (End - Ani) * (End - Ani);
508  }
509  break;
510  case 4: // 감가변속
511  if(AniFrame <= EndFrame / 2)
512  {
513  const int End = EndFrame / 2;
514  const int EndHalf = End * End;
515  EndValue = EndHalf * 2;
516  AniValue = EndHalf - (End - AniFrame) * (End - AniFrame);
517  }
518  else
519  {
520  const int Ani = AniFrame - EndFrame / 2;
521  const int End = EndFrame - EndFrame / 2;
522  const int EndHalf = End * End;
523  EndValue = EndHalf * 2;
524  AniValue = EndHalf + Ani * Ani;
525  }
526  break;
527  }
528  // 이동적용
529  BxAssert("BxTween", EndValue != 0);
530  AnimationPtr[i].AnimationMoverX = (AnimationPtr[i].AnimationTargetX - AnimationPtr[i].MoverX)
531  * AniValue / EndValue + AnimationPtr[i].MoverX;
532  AnimationPtr[i].AnimationMoverY = (AnimationPtr[i].AnimationTargetY - AnimationPtr[i].MoverY)
533  * AniValue / EndValue + AnimationPtr[i].MoverY;
534  // 중력상황
535  if(BxUtil::GetValue(GravityMin, GravityMax, i))
536  {
537  AnimationPtr[i].AnimationGravityVector += (int)((((huge) BxUtil::GetValue(GravityMin, GravityMax, i)) << 16) * MsPerFrame / 1000);
538  AnimationPtr[i].AnimationTargetAddY += AnimationPtr[i].AnimationGravityVector;
539  }
540  // 바람상황
541  if(BxUtil::GetValue(WindMin, WindMax, i))
542  {
543  AnimationPtr[i].AnimationTargetAddX += (int)(((huge)(BxUtil::Sin(AnimationPtr[i].AnimationWindAngle * 1024 / 360)))
544  * BxUtil::GetValue(WindMin, WindMax, i) * MsPerFrame / 1000);
545  AnimationPtr[i].AnimationTargetAddY -= (int)(((huge)(BxUtil::Cos(AnimationPtr[i].AnimationWindAngle * 1024 / 360)))
546  * BxUtil::GetValue(WindMin, WindMax, i) * MsPerFrame / 1000);
547  }
548  // 타겟적용
549  if(BxUtil::GetValue(GravityMin, GravityMax, i) || BxUtil::GetValue(WindMin, WindMax, i))
550  AnimationPtr[i].SetTarget((AnimationPtr[i].AnimationTargetAddX + 0x7FFF) >> 16, (AnimationPtr[i].AnimationTargetAddY + 0x7FFF) >> 16);
551  // 인력적용
552  if(Position == 19 || (Position % 9) / 3 == 2)
553  AnimationPtr[i].PullTarget(AniFrame, EndFrame);
554  // 불투명도적용
555  AnimationPtr[i].SetOpacity(BxUtil::GetValue(OpacityMin, OpacityMax, i), AniFrame, EndFrame);
556  // 진행패턴계산
557  if(0 < Pattern)
558  {
559  BxAssert("BxTween", EndFrame != 0);
560  int MoverAddX = 0, MoverAddY = 0, CalcValue1 = 0, CalcValue2 = 0;
561  switch(Pattern)
562  {
563  case 1: // 시계
564  CalcValue1 = (1024 / 2) * (AnimationPtr[i].AnimationCalcValue & 0xFF);

601  }
